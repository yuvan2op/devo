pipeline {
    agent any

    environment {
        IMAGE_NAME = "yuvan2op/foodtestweb"
    }

    stages {
        stage('Checkout Code') {
            steps {
                git branch: 'main', url: 'https://github.com/yuvan2op/foodtestweb.git'
            }
        }

        stage('Build Docker Image') {
            steps {
                sh 'docker build -t $IMAGE_NAME .'
            }
        }

        stage('Run Container') {
            steps {
                sh 'docker stop foodtestweb || true && docker rm foodtestweb || true'
                sh 'docker run -d -p 8080:80 --name foodtestweb $IMAGE_NAME'
            }
        }
    }
}
